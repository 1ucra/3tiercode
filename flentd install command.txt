sudo yum update -y

curl -L https://toolbelt.treasuredata.com/sh/install-amazon2-td-agent4.sh | sudo sh

sudo systemctl start td-agent
sudo systemctl enable td-agent
sudo systemctl status td-agent

sudo td-agent-gem install fluent-plugin-s3

sudo vi /etc/td-agent/td-agent.conf

sudo systemctl restart td-agent

sudo vi /etc/nginx/nginx.conf 에서 로그 생성 설정

    access_log /var/log/nginx/access.log main;
    error_log /var/log/nginx/error.log;
    include /etc/nginx/conf.d/*.conf;

sudo systemctl restart nginx
curl http://localhost - 로그 생성

sudo vi /etc/td-agent/td-agent.conf td에서 파일 출력 설정

<source>
  @type tail
  path /var/log/nginx/access.log
  pos_file /var/log/td-agent/nginx-access.log.pos
  tag nginx.access
  <parse>
    @type nginx
  </parse>
</source>
<match nginx.*>
  @type elasticsearch
  @id elasticsearch
  host "a4ced084d56264a6ea0754761d650776-828082069.ap-northeast-2.elb.amazonaws.com"
  port 9200
  logstash_format true
  logstash_prefix ktd-logstash
  include_tag_key true
  tag_key "@log_name"
  flush_interval 10s
  <buffer>
    flush_interval 10s
  </buffer>
</match>


sudo systemctl restart td-agent
